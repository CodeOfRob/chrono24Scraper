   function storeSearchStorageModel() {
      if('searchStorageModel' in c24.meta.data) {

         var newEntry = c24.meta.data.searchStorageModel;

         try {
            if(localStorage.getItem('searchStorage') != null) {
               searchStorage = JSON.parse(localStorage.getItem('searchStorage'));

               searchStorage.forEach(function(entry, index) {
                  if(newEntry.label === entry.label && newEntry.explanation === entry.explanation) {
                     searchStorage.splice(index, 1);
                  }
               });
            }

            if(String(newEntry.label).trim().length) {
               searchStorage.unshift(newEntry);
            }

            searchStorage = searchStorage.slice(0, 5);

         }
      }

   function shareByMail() {

   function shareByFacebook(shareUrl) {
      tce(c24.meta.pageId, c24.getData('share-popover-facebook-tracking-action'));

      var encodedUrl = encodeURIComponent(shareUrl);

      c24.tools.newWindow('https://www.facebook.com/sharer.php?u=' + encodedUrl, 'facebook', 550, 420, 'yes');
   }

   function shareByTwitter(shareUrl) {
      tce(c24.meta.pageId, c24.getData('share-popover-twitter-tracking-action'));

      var encodedUrl = encodeURIComponent(shareUrl);
      var encodedText = encodeURIComponent(c24.getData('share-popover-text') + c24.getData('share-popover-twitter-suffix') + ' ...');

      c24.tools.newWindow('https://twitter.com/share?url=' + encodedUrl + '&text=' + encodedText, 'twitter', 550, 420, 'no');
   }

   function shareByWhatsapp(shareUrl) {
      tce(c24.meta.pageId, 'iconbar-whatsapp');

      var text = c24.getData('share-popover-text');

      c24.tools.newWindow('whatsapp://send?text=' + encodeURIComponent(text + ' ' + shareUrl), 'whatsapp', 550, 420, 'no');
   }

   function shareByCopyLink() {
      tce(c24.meta.pageId, c24.getData('share-popover-copy-tracking-action'));
   }

   function openSharingPopover() {
      c24.colorbox.create('jq-popover-share', {
         size: "sm", onComplete: function () {
            tce(c24.meta.pageId, c24.getData('share-popover-click-event'));

            var clipboard = new Clipboard('.shortLinkButton', {
               container: document.querySelector('#colorbox #jq-popover-share')
            });

            clipboard.on('success', function (e) {
               $('.shortLinkButton').addClass('copied');
            });

            $('#cboxClose').on('click', function () {
               $('.shortLinkButton').removeClass('copied');
            });
         }
      });
   }

   function initSharePopover () {
      $('.share-offer').on('click', function () {
         if (c24.meta.isC) {
            return;
         }

         var relayUrl = c24.getData('share-popover-relay-url');

         if (relayUrl) {
            axios.post(relayUrl, c24.meta.csrf)
               .then(function (data) {
                  var encodedSubject = encodeURIComponent(c24.getData('share-popover-subject'));
                  var encodedText = encodeURIComponent(c24.getData('share-popover-text') + ' ' + data.data.shareUrl);

                  $('.shortLink').val(data.data.shareUrl);
                  $('.shortLinkButton').attr('data-clipboard-text', data.data.shareUrl);
                  $('#jq-popover-share .js-share-link').data('share-url', data.data.shareUrl);
                  $('.js-share-mail').attr('href', 'mailto:?subject=' + encodedSubject + '&body=' + encodedText);

                  openSharingPopover();
               })

         } else {
            openSharingPopover();
         }
      });
   }
